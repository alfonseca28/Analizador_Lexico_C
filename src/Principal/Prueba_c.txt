/*En una tienda de decuento se efectua una promocion, en la cual se hace un descuento sobre el
valor de la compra total, segun el numero del cliente saque al pagar en caja, si la bolita es 1
no se le hara ningun descuento, si la bolita es 2 se le hara un 10%, si la bolita es 3 un 25%,
si es 4 el 50% y si es 5 es el 100%.*/
#include<stdio.h>

int main()
{
	int num_b;
	float d, paso1, valor_c;

	printf("Introduce el valor total de la compra");
	scanf("%f",&valor_c);
	printf("Introduce el numero de la bolita que conseguiste: ");
	scanf("%d",&num_b);
	switch (num_b)
	{
	case 1:
		printf("Lo sentimos, no se te ha aplicado ningun descuento.");
		printf("Pagas el total de tu compra:");
		printf("%.2f\n", valor_c);
		break;
	case 2:
		paso1 = valor_c*0.10;
		d = valor_c-paso1;
		printf("Felicidades, has obtenido un 10%% de descuento.");
		printf("En total pagas:");
		printf("%.2f\n",d);
		break;
	case 3:
		paso1 = valor_c*0.25;
		d = valor_c-paso1;
		printf("Felicidades, has obtenido un 25%% de descuento.");
		printf("En total pagas:");
		printf("%.2f\n",d);
		break;
	case 4:
		paso1 = valor_c*0.50;
		d = valor_c-paso1;
		printf("Felicidades, has obtenido un 50%% de descuento.");
		printf("En total pagas: ");
		printf("%.2f\n",d);
		break;
	case 5:
		paso1 = valor_c*1;
		d = valor_c-paso1;
		printf("Felicidades, has obtenido un 100%% de descuento.");
		printf("En total pagas:");
		printf("%.2f\n",d);
		break;
	}
	return 0;
}
