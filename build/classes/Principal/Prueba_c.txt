#include <stdio.h>
main(){
	int c=1,n1=0,n2=0,n3=0;
	float s1=0,s2=0,s3=0,n=0,total=0;
	while (c==1){
		printf ("De cuanto fue el gasto");
		scanf ("%f",&n);
		if (n<=500){
			n1=n1+1;
			s1=s1+n;
		}
		else
			if (n>500 && n<=1000){
				n2=n2+1;
				s2=s2+n;
			}
			if (n>1000){
				n3=n3+1;
				s3=s3+n;
			}
		printf ("Hubo otro gasto 1.SI 2.NO");
		scanf ("%d",&c);
	}
	total=s1+s2+s3;
	printf ("Los gastos menores o iguales a 500 fueron %d y su costo fue de %.2f ",n1,s1);
	printf ("Los gastos mayores a 500 y menores o iguales a 1000 fueron %d y su costo fue de %.2f",n2,s2);
	printf ("Los gatos mayores a 1000 fueron %d y su costo fue de %.2f",n3,s3);
	printf ("El gasto global fue de %.2f",total);
	return 0;
}
